// we will do some ajax post request and hence we will need to grab csrf from cookie. // https://docs.djangoproject.com/en/1.11/ref/csrf/#ajax function getCookie(name) { var cookieValue = null; if (document.cookie && document.cookie !== '') { var cookies = document.cookie.split(';'); for (var i = 0; i < cookies.length; i++) { var cookie = jQuery.trim(cookies[i]); // Does this cookie string begin with the name we want? if (cookie.substring(0, name.length + 1) === (name + '=')) { cookieValue = decodeURIComponent(cookie.substring(name.length + 1)); break; } } } return cookieValue; } var csrftoken = getCookie('csrftoken'); $.ajaxSetup({ headers: { "X-CSRFToken": csrftoken } }); // end of csrftoken cookie set up
